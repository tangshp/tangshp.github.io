实现一个字符设备驱动

字符串设备驱动的框架

字符串设备驱动的框架包括以下三个函数：

驱动的入口函数是module_init(fun_init)
驱动的出口函数是module_exit(fun_exit)
驱动的版权函数是MODULE_LICENSE("GPL")

入口函数

驱动的入口函数module_init(fun_init)负责在内核中申请设备号，定义设备类型对象，设置该设备对应的打开，读，写等操作；并把设备号与设备对象关联起来，把设备对象与文件操作关联起来；

  • 在内核中申请设备号

    // reg_major是主设备号，reg_minor是次设备号
// 用主次设备号生成一个设备号devNum
devNum = MKDEV(reg_major, reg_minor);
// 在内核中注册该设备号，在内核中唯一标识了一个id，该id在内核中不能被其他设备暂用
// subDevNum表示目前开始注册第subDevNum个设备号
if(OK == register_chrdev_region(devNum, subDevNum, "helloworld")){
    printk(KERN_EMERG"register_chrdev_region ok \n"); 
}else {
printk(KERN_EMERG"register_chrdev_region error n");
    return ERROR;
}
printk(KERN_EMERG" hello driver init \n");

  • 定义字符设备类型对象

    // cdev表示一个字符设备
gDev = kzalloc(sizeof(struct cdev), GFP_KERNEL);

  • 定义相关操作
    linux把设备也抽象成文件，所以要定义文件的读写相关操作

    // linux中设备也是文件，可以用file_operations来定义一些对文件操作的函数
gFile = kzalloc(sizeof(struct file_operations), GFP_KERNEL);
// 对要制作的字符串设备驱动定义 打开，读，写的操作函数
gFile->open = hello_open;
gFile->read = hello_read;
gFile->write = hello_write;
gFile->owner = THIS_MODULE;

  • 把设备号与设备对象关联起来，把设备对象与文件操作关联起来

    // 字符设备初始化，把字符设备驱动和该驱动相关的操作函数联系起来
cdev_init(gDev, gFile);
// 设定该字符驱动的在内核中的设备号，1表示我们注册了一个设备
cdev_add(gDev, devNum, 1);

  • 操作函数的定义

    int hello_open(struct inode *p, struct file *f)

    printk(KERN_EMERG"hello_open\r\n");
    return 0;

出口函数

驱动的出口函数是module_exit(fun_exit)负责卸载该设备的相关操作

printk(KERN_EMERG"unregister_chrdev_region ok \n"); 
cdev_del(gDev);
unregister_chrdev_region(devNum, subDevNum);

许可协议

GPL说明

MODULE_LICENSE("GPL");

源文件

#include <linux/module.h>
#include <linux/moduleparam.h>
#include <linux/cdev.h>
#include <linux/fs.h>
#include <linux/wait.h>
#include <linux/poll.h>
#include <linux/sched.h>
#include <linux/slab.h>

#define BUFFER_MAX    (10)
#define OK            (0)
#define ERROR         (-1)

struct cdev *gDev;
struct file_operations *gFile;
dev_t  devNum;
unsigned int subDevNum = 1;
int reg_major  =  232;    
int reg_minor =   0;
char *buffer;
int flag = 0;
int hello_open(struct inode *p, struct file *f)
{
    printk(KERN_EMERG"hello_open\r\n");
    return 0;
}

ssize_t hello_write(struct file *f, const char __user *u, size_t s, loff_t *l)
{
    printk(KERN_EMERG"hello_write\r\n");
    return 0;
}
ssize_t hello_read(struct file *f, char __user *u, size_t s, loff_t *l)
{
    printk(KERN_EMERG"hello_read\r\n");      
    return 0;
}
int hello_init(void)
{
    // reg_major是主设备号，reg_minor是次设备号
    // 用主次设备号生成一个设备号devNum
    devNum = MKDEV(reg_major, reg_minor);
	// 在内核中注册该设备号，在内核中唯一标识了一个id，该id在内核中不能被其他设备暂用
	// subDevNum表示目前开始注册第subDevNum个设备号
    if(OK == register_chrdev_region(devNum, subDevNum, "helloworld")){
        printk(KERN_EMERG"register_chrdev_region ok \n"); 
    }else {
    printk(KERN_EMERG"register_chrdev_region error n");
        return ERROR;
    }
    printk(KERN_EMERG" hello driver init \n");

	// cdev表示一个字符设备
    gDev = kzalloc(sizeof(struct cdev), GFP_KERNEL);

	// linux中设备也是文件，可以用file_operations来定义一些对文件操作的函数
    gFile = kzalloc(sizeof(struct file_operations), GFP_KERNEL);
	// 对要制作的字符串设备驱动定义 打开，读，写的操作函数
    gFile->open = hello_open;
    gFile->read = hello_read;
    gFile->write = hello_write;
    gFile->owner = THIS_MODULE;

	// 字符设备初始化，把字符设备驱动和该驱动相关的操作函数联系起来
    cdev_init(gDev, gFile);
	// 设定该字符驱动的在内核中的设备号，1表示我们注册了一个设备
    cdev_add(gDev, devNum, 1);
    return 0;
}

void __exit hello_exit(void)
{
	printk(KERN_EMERG"unregister_chrdev_region ok \n"); 
    cdev_del(gDev);
    unregister_chrdev_region(devNum, subDevNum);
    return;
}
module_init(hello_init);
module_exit(hello_exit);
MODULE_LICENSE("GPL");

Makefile文件

ifneq ($(KERNELRELEASE),)
obj-m := helloDev.o
else
PWD := $(shell pwd)
#KDIR:= /lib/modules/4.4.0-31-generic/build
KDIR := /lib/modules/`uname -r`/build
all:
	make -C $(KDIR) M=$(PWD)
clean:	
	rm -rf *.o *.ko *.mod.c *.symvers *.c~ *~
endif
  • Makefile文件说明
    • 执行Makefile时, KERNELRELEASE是空, 所以执行到了else;
    • PWD变量是为了二次执行该Makefile;
    • KDIR是内核开发包的路径，在该路径执行玩Makefile后, KERNELRELEASE变成非空
    • 参数M表示在路径KDIR下执行完后, 回到PWD路径执行Makefile
    • 回到当前路径二次执行Makefile时, 内核会收集所有obj-m对象, 把源文件加工为.o文件；在把.o文件加工为.ko文件

.ko文件是驱动文件
